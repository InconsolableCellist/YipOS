]

  (-2.4, -2.0) rectangle (2.4, 2.2);

\node[font=\tiny\ttfamily, crtgreen] at (0, 2.0) {Render Texture};

% Zone center dots (9 touch buttons)
\foreach \cx/\cy in {-1.5/1.25, 0/1.25, 1.5/1.25,

                     -1.5/-1.25, 0/-1.25, 1.5/-1.25} {
  \node[contact] at (\cx, \cy) {};
}

% === 6 CONTACT SENDERS (hardware) at row endpoints ===
% Small labels at the 6 physical sender positions
\node[oscparam, left=3pt] at (-1.5, 1.25) {\_11};

% === OSC OUTPUT ARROW ===

\end{tikzpicture}
\caption{The Yip-Boi PDA and its contact points. Nine touchscreen zones
(3$\times$3 grid) are defined by six Contact Senders (\texttt{CRT\_Wrist\_XY});
five physical button contacts detect button presses. All contacts send OSC data
to the Yip~OS companion app. The CRT display is fed by the Render Texture
produced by the Williams Tube (Fig.~\ref{fig:williams-tube}).}
\label{fig:yipboi-contacts}
\end{figure}
